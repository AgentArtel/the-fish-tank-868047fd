// ArticleDetail.tsx — REFERENCE (Vite + React + TS) — routes: /blog/:slug and /guides/:slug
// Wired to getArticleBySlug(). Renders body (Markdown), author byline (E-E-A-T), inline FAQs,
// related products, + Article/Person + FAQPage JSON-LD. Rename .txt→.tsx, fix imports.
//
// Markdown: install a renderer (e.g. `react-markdown`) — shown as <Markdown>.

import { useQuery } from "@tanstack/react-query";
import { useParams, Link } from "react-router-dom";
import { Helmet } from "react-helmet-async";
import Markdown from "react-markdown";
import { getArticleBySlug, listProducts } from "@/data/client/tft-data";
import { ProductCard } from "@/components/ds";

const SITE = "https://thefishtank.com";

export default function ArticleDetail({ kind = "post" }: { kind?: "post" | "guide" }) {
  const { slug = "" } = useParams();
  const base = kind === "guide" ? "/guides" : "/blog";

  const { data: article, isLoading, isError } = useQuery({
    queryKey: ["article", slug],
    queryFn: () => getArticleBySlug(slug),
    enabled: !!slug,
  });

  // related products by slug (article carries relatedProductSlugs)
  const { data: related } = useQuery({
    queryKey: ["article-related", slug],
    queryFn: async () => {
      const out = await Promise.all((article!.relatedProductSlugs || []).slice(0, 4).map(async (s: string) => {
        try { const { getProductBySlug } = await import("@/data/client/tft-data"); return await getProductBySlug(s); } catch { return null; }
      }));
      return out.filter(Boolean);
    },
    enabled: !!article?.relatedProductSlugs?.length,
  });

  if (isLoading) return <Skeleton />;
  if (isError || !article) return <NotFound base={base} />;

  return (
    <article style={{ maxWidth: 760, margin: "0 auto", padding: "var(--space-10) var(--gutter) 0" }}>
      <ArticleSeo article={article} base={base} />

      <nav style={{ display: "flex", alignItems: "center", gap: "var(--space-2)", font: "var(--type-small)", color: "var(--text-muted)", marginBottom: "var(--space-5)" }}>
        <Link to="/">Home</Link>
        <i data-lucide="chevron-right" style={{ width: 14, height: 14 }} />
        <Link to={base}>{kind === "guide" ? "Guides" : "Blog"}</Link>
      </nav>

      {!!article.topics?.length && <div className="tft-eyebrow" style={{ marginBottom: 10 }}>{article.topics[0]}</div>}
      <h1 style={{ font: "var(--fw-extra) var(--text-4xl)/1.05 var(--font-display)" }}>{article.title}</h1>

      {/* byline — E-E-A-T */}
      <div style={{ display: "flex", alignItems: "center", gap: "var(--space-3)", margin: "var(--space-5) 0 var(--space-6)" }}>
        {article.author?.avatarUrl && <img src={article.author.avatarUrl} alt={article.author.name} style={{ width: 44, height: 44, borderRadius: "var(--radius-full)", objectFit: "cover" }} />}
        <div>
          {article.author?.name && <div style={{ font: "var(--fw-semibold) var(--text-sm)/1.2 var(--font-sans)", color: "var(--text-heading)" }}>{article.author.name}</div>}
          <div style={{ font: "var(--type-caption)", color: "var(--text-muted)" }}>
            {article.author?.title && <span>{article.author.title} · </span>}
            {fmtDate(article.publishedAt)}{article.readingMinutes ? ` · ${article.readingMinutes} min read` : ""}
          </div>
        </div>
      </div>

      {article.heroImage?.url && (
        <img src={article.heroImage.url} alt={article.heroImage.alt ?? article.title}
          style={{ width: "100%", borderRadius: "var(--radius-xl)", boxShadow: "var(--shadow-md)", marginBottom: "var(--space-8)" }} />
      )}

      {/* body */}
      <div className="prose" style={{ font: "var(--type-body)", color: "var(--text-body)", lineHeight: "var(--leading-relaxed)" }}>
        <Markdown>{article.bodyMarkdown}</Markdown>
      </div>

      {/* inline FAQ */}
      {!!article.faqs?.length && (
        <section style={{ marginTop: "var(--space-12)" }}>
          <h2 style={{ font: "var(--type-h3)", marginBottom: "var(--space-5)" }}>Frequently asked</h2>
          {article.faqs.map((f: any) => (
            <details key={f.id} style={{ borderBottom: "1px solid var(--border-default)", padding: "var(--space-4) 0" }}>
              <summary style={{ font: "var(--fw-semibold) var(--text-base)/1.4 var(--font-sans)", color: "var(--text-heading)", cursor: "pointer" }}>{f.question}</summary>
              <div style={{ font: "var(--type-body)", color: "var(--text-secondary)", marginTop: "var(--space-2)" }}><Markdown>{f.answerMarkdown}</Markdown></div>
            </details>
          ))}
        </section>
      )}

      {/* related products — cross-link to catalog */}
      {!!related?.length && (
        <section style={{ margin: "var(--space-16) 0" }}>
          <h2 style={{ font: "var(--type-h3)", marginBottom: "var(--space-6)" }}>Shop this guide</h2>
          <div style={{ display: "grid", gridTemplateColumns: "repeat(4, 1fr)", gap: "var(--space-5)" }}>
            {related.map((p: any) => (
              <Link key={p.id} to={`/products/${p.slug}`} style={{ textDecoration: "none" }}>
                <ProductCard image={p.images[0]?.url} vendor={p.originRegion} name={p.name} scientificName={p.scientificName} price={p.price} compareAt={p.compareAtPrice} stock={p.availability === "sold" ? "sold" : "live"} />
              </Link>
            ))}
          </div>
        </section>
      )}
    </article>
  );
}

function ArticleSeo({ article, base }: any) {
  const url = `${SITE}${base}/${article.slug}`;
  const desc = (article.seo?.description || article.excerpt || article.title).slice(0, 155);
  const img = article.seo?.ogImage || article.heroImage?.url;
  const graph: any[] = [{
    "@context": "https://schema.org",
    "@type": article.kind === "guide" ? "HowTo" : "Article",
    headline: article.title,
    image: img ? [img] : undefined,
    datePublished: article.publishedAt,
    dateModified: article.updatedAt || article.publishedAt,
    author: article.author ? { "@type": "Person", name: article.author.name, jobTitle: article.author.title || undefined } : { "@type": "Organization", name: "The Fish Tank" },
    publisher: { "@type": "Organization", name: "The Fish Tank", logo: { "@type": "ImageObject", url: `${SITE}/assets/logo-fish.png` } },
    mainEntityOfPage: url,
  }];
  if (article.faqs?.length) graph.push({
    "@context": "https://schema.org", "@type": "FAQPage",
    mainEntity: article.faqs.map((f: any) => ({ "@type": "Question", name: f.question, acceptedAnswer: { "@type": "Answer", text: f.answerMarkdown } })),
  });
  return (
    <Helmet>
      <title>{`${article.seo?.title || article.title} | The Fish Tank`}</title>
      <meta name="description" content={desc} />
      <link rel="canonical" href={article.seo?.canonical || url} />
      {article.seo?.noindex && <meta name="robots" content="noindex" />}
      <meta property="og:type" content="article" />
      <meta property="og:title" content={article.title} />
      <meta property="og:description" content={desc} />
      {img && <meta property="og:image" content={img} />}
      {graph.map((g, i) => <script key={i} type="application/ld+json">{JSON.stringify(g)}</script>)}
    </Helmet>
  );
}

const fmtDate = (d: string) => d ? new Date(d).toLocaleDateString("en-US", { year: "numeric", month: "long", day: "numeric" }) : "";

function Skeleton() {
  return <div style={{ maxWidth: 760, margin: "0 auto", padding: "var(--space-12) var(--gutter)", display: "flex", flexDirection: "column", gap: "var(--space-4)" }}>
    {[40, 80, 30, "16/9", 90, 90, 70].map((w, i) => typeof w === "string"
      ? <div key={i} style={{ aspectRatio: w, background: "var(--surface-sunken)", borderRadius: "var(--radius-xl)" }} />
      : <div key={i} style={{ height: 18, width: `${w}%`, borderRadius: 6, background: "var(--surface-sunken)" }} />)}
  </div>;
}
function NotFound({ base }: any) {
  return <div style={{ maxWidth: 560, margin: "0 auto", padding: "var(--space-24) var(--gutter)", textAlign: "center" }}>
    <h1 style={{ font: "var(--type-h2)" }}>Article not found</h1>
    <p style={{ font: "var(--type-lead)", color: "var(--text-secondary)", marginTop: "var(--space-3)" }}>It may have moved.</p>
    <Link to={base} style={{ color: "var(--brand-primary)", fontWeight: 600 }}>Back to all articles</Link>
  </div>;
}
