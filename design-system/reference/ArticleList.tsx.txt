// ArticleList.tsx — REFERENCE (Vite + React + TS) — routes: /blog and /guides
// One component, two routes: pass kind="post" (/blog) or kind="guide" (/guides) via the route element.
// Wired to listArticles(). Rename .txt→.tsx, fix imports, swap data-lucide for lucide-react.

import { useState } from "react";
import { useQuery, keepPreviousData } from "@tanstack/react-query";
import { Link } from "react-router-dom";
import { Helmet } from "react-helmet-async";
import { listArticles } from "@/data/client/tft-data";

const SITE = "https://thefishtank.com";
const PAGE = 12;

export default function ArticleList({ kind = "post" }: { kind?: "post" | "guide" }) {
  const [page, setPage] = useState(0);
  const cfg = kind === "guide"
    ? { path: "/guides", title: "Reef Care Guides", blurb: "Practical, Utah-tested guides from our reef team — acclimation, water chemistry, livestock care, and tank setup." }
    : { path: "/blog", title: "The Fish Tank Blog", blurb: "New arrivals, livestock highlights, events, and reef news from our Sandy showroom." };

  const { data, isLoading } = useQuery({
    queryKey: ["articles", kind, page],
    queryFn: () => listArticles({ kind, limit: PAGE, offset: page * PAGE }),
    placeholderData: keepPreviousData,
  });
  const articles = data?.articles ?? [];

  return (
    <div>
      <Helmet>
        <title>{`${cfg.title} | The Fish Tank — Sandy, UT`}</title>
        <meta name="description" content={cfg.blurb} />
        <link rel="canonical" href={`${SITE}${cfg.path}`} />
      </Helmet>

      <section style={{ background: "var(--grad-ocean)", color: "var(--text-on-ocean)" }}>
        <div style={{ maxWidth: "var(--container-xl)", margin: "0 auto", padding: "var(--space-14) var(--gutter)" }}>
          <h1 style={{ font: "var(--fw-extra) var(--text-5xl)/1 var(--font-display)", color: "#fff", margin: 0 }}>{cfg.title}</h1>
          <p style={{ font: "var(--type-lead)", color: "var(--text-on-ocean-muted)", marginTop: "var(--space-3)", maxWidth: 580 }}>{cfg.blurb}</p>
        </div>
      </section>

      <section style={{ maxWidth: "var(--container-xl)", margin: "0 auto", padding: "var(--space-12) var(--gutter)" }}>
        {isLoading ? (
          <Grid>{Array.from({ length: 6 }).map((_, i) => <CardSkeleton key={i} />)}</Grid>
        ) : articles.length === 0 ? (
          <div style={{ textAlign: "center", padding: "var(--space-20)", color: "var(--text-muted)" }}>
            <i data-lucide="feather" style={{ width: 40, height: 40 }} />
            <div style={{ font: "var(--type-h4)", color: "var(--text-heading)", marginTop: 12 }}>Fresh {kind === "guide" ? "guides" : "posts"} coming soon</div>
            <p style={{ font: "var(--type-small)", marginTop: 6 }}>Check back shortly — our reef team is writing.</p>
          </div>
        ) : (
          <Grid>
            {articles.map((a: any) => (
              <Link key={a.id} to={`${cfg.path}/${a.slug}`} style={{ textDecoration: "none" }}>
                <ArticleCard article={a} />
              </Link>
            ))}
          </Grid>
        )}
      </section>
    </div>
  );
}

function ArticleCard({ article: a }: any) {
  return (
    <article style={{ background: "var(--surface-card)", borderRadius: "var(--radius-lg)", overflow: "hidden", boxShadow: "var(--ring-hairline), var(--shadow-sm)", height: "100%" }}>
      <div style={{ aspectRatio: "16/10", background: "var(--surface-sunken)", overflow: "hidden" }}>
        {a.heroImage?.url && <img src={a.heroImage.url} alt={a.heroImage.alt ?? a.title} loading="lazy" style={{ width: "100%", height: "100%", objectFit: "cover" }} />}
      </div>
      <div style={{ padding: "var(--space-5)" }}>
        {!!a.topics?.length && <div className="tft-eyebrow" style={{ marginBottom: 8 }}>{a.topics[0]}</div>}
        <h2 style={{ font: "var(--fw-bold) var(--text-xl)/1.2 var(--font-display)", color: "var(--text-heading)" }}>{a.title}</h2>
        {a.excerpt && <p style={{ font: "var(--type-small)", color: "var(--text-secondary)", marginTop: 8 }}>{a.excerpt}</p>}
        <div style={{ display: "flex", alignItems: "center", gap: "var(--space-2)", marginTop: "var(--space-4)", font: "var(--type-caption)", color: "var(--text-muted)" }}>
          {a.author?.name && <span>{a.author.name}</span>}
          {a.readingMinutes && <span>· {a.readingMinutes} min read</span>}
        </div>
      </div>
    </article>
  );
}

function Grid({ children }: any) {
  return <div style={{ display: "grid", gridTemplateColumns: "repeat(3, 1fr)", gap: "var(--space-6)" }}>{children}</div>;
}
function CardSkeleton() {
  return <div style={{ borderRadius: "var(--radius-lg)", overflow: "hidden", boxShadow: "var(--ring-hairline)" }}>
    <div style={{ aspectRatio: "16/10", background: "var(--surface-sunken)" }} />
    <div style={{ padding: "var(--space-5)", display: "flex", flexDirection: "column", gap: 8 }}>
      {[50, 90, 70].map((w, i) => <div key={i} style={{ height: 12, width: `${w}%`, borderRadius: 4, background: "var(--surface-sunken)" }} />)}
    </div>
  </div>;
}
