// Faq.tsx — REFERENCE (Vite + React + TS) — route: /faq
// Wired to listFaqs(). Accordion grouped by category + FAQPage JSON-LD (rich results).
// Rename .txt→.tsx, fix imports, swap data-lucide for lucide-react. Markdown via react-markdown.

import { useMemo, useState } from "react";
import { useQuery } from "@tanstack/react-query";
import { Helmet } from "react-helmet-async";
import Markdown from "react-markdown";
import { listFaqs } from "@/data/client/tft-data";

const SITE = "https://thefishtank.com";

export default function Faq() {
  const { data: faqs = [], isLoading } = useQuery({ queryKey: ["faqs"], queryFn: () => listFaqs() });

  // group by category, preserving sortOrder
  const groups = useMemo(() => {
    const by: Record<string, any[]> = {};
    [...faqs].sort((a, b) => a.sortOrder - b.sortOrder).forEach((f) => { (by[f.category || "General"] ||= []).push(f); });
    return Object.entries(by);
  }, [faqs]);

  return (
    <div>
      <Helmet>
        <title>FAQ — The Fish Tank | Shipping, Livestock &amp; Reef Rewards</title>
        <meta name="description" content="Answers about livestock shipping, the arrival guarantee, Reef Rewards, and visiting our Sandy, UT reef showroom." />
        <link rel="canonical" href={`${SITE}/faq`} />
        {faqs.length > 0 && (
          <script type="application/ld+json">{JSON.stringify({
            "@context": "https://schema.org", "@type": "FAQPage",
            mainEntity: faqs.map((f: any) => ({ "@type": "Question", name: f.question, acceptedAnswer: { "@type": "Answer", text: f.answerMarkdown } })),
          })}</script>
        )}
      </Helmet>

      <section style={{ background: "var(--grad-ocean)", color: "var(--text-on-ocean)" }}>
        <div style={{ maxWidth: "var(--container-xl)", margin: "0 auto", padding: "var(--space-14) var(--gutter)" }}>
          <h1 style={{ font: "var(--fw-extra) var(--text-5xl)/1 var(--font-display)", color: "#fff", margin: 0 }}>Frequently asked questions</h1>
          <p style={{ font: "var(--type-lead)", color: "var(--text-on-ocean-muted)", marginTop: "var(--space-3)", maxWidth: 560 }}>
            Shipping, the arrival guarantee, Reef Rewards, and visiting the shop. Still stuck? Call us at (801) 887-7000.
          </p>
        </div>
      </section>

      <section style={{ maxWidth: 820, margin: "0 auto", padding: "var(--space-12) var(--gutter)" }}>
        {isLoading ? (
          [0, 1, 2, 3].map((i) => <div key={i} style={{ height: 56, borderRadius: "var(--radius-md)", background: "var(--surface-sunken)", marginBottom: "var(--space-3)" }} />)
        ) : groups.length === 0 ? (
          <p style={{ textAlign: "center", color: "var(--text-muted)", padding: "var(--space-16)" }}>FAQ coming soon.</p>
        ) : (
          groups.map(([category, items]) => (
            <div key={category} style={{ marginBottom: "var(--space-10)" }}>
              <h2 style={{ font: "var(--type-h4)", color: "var(--text-heading)", marginBottom: "var(--space-4)" }}>{category}</h2>
              <div>{items.map((f) => <FaqItem key={f.id} f={f} />)}</div>
            </div>
          ))
        )}
      </section>
    </div>
  );
}

function FaqItem({ f }: any) {
  const [open, setOpen] = useState(false);
  return (
    <div style={{ borderBottom: "1px solid var(--border-default)" }}>
      <button onClick={() => setOpen((o) => !o)}
        style={{ width: "100%", display: "flex", alignItems: "center", justifyContent: "space-between", gap: "var(--space-4)", padding: "var(--space-4) 0", background: "none", border: "none", cursor: "pointer", textAlign: "left" }}>
        <span style={{ font: "var(--fw-semibold) var(--text-base)/1.4 var(--font-sans)", color: "var(--text-heading)" }}>{f.question}</span>
        <i data-lucide={open ? "minus" : "plus"} style={{ width: 18, height: 18, color: "var(--brand-primary)", flex: "none" }} />
      </button>
      {open && (
        <div style={{ font: "var(--type-body)", color: "var(--text-secondary)", padding: "0 0 var(--space-4)", lineHeight: "var(--leading-relaxed)" }}>
          <Markdown>{f.answerMarkdown}</Markdown>
        </div>
      )}
    </div>
  );
}
